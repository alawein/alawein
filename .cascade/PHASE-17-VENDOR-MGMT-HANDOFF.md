### 2. Create Vendor Status Checker (10 min)

Create `tools/vendors/status.ts`:

```typescript
#!/usr/bin/env tsx
/**
 * Vendor status checker
 */
import { readFileSync, existsSync } from 'fs';
import { load } from 'js-yaml';

const VENDOR_REGISTRY = '.metaHub/vendors/registry.yaml';

// Status page URLs for common vendors
const STATUS_PAGES: Record<string, string> = {
  Supabase: 'https://status.supabase.com',
  Stripe: 'https://status.stripe.com',
  Vercel: 'https://www.vercel-status.com',
  GitHub: 'https://www.githubstatus.com',
  OpenAI: 'https://status.openai.com',
  Anthropic: 'https://status.anthropic.com',
};

interface Vendor {
  name: string;
  category: string;
  sla: string;
  products_using: string[];
}

function showVendors() {
  if (!existsSync(VENDOR_REGISTRY)) {
    console.error('‚ùå Vendor registry not found');
    return;
  }

  const config = load(readFileSync(VENDOR_REGISTRY, 'utf-8')) as any;

  console.log('\nü§ù VENDOR REGISTRY\n');
  console.log('='.repeat(60));

  console.log('\nüî¥ CRITICAL VENDORS:');
  config.vendors.critical.forEach((v: Vendor) => {
    console.log(
      `  ${v.name.padEnd(15)} | ${v.category.padEnd(12)} | SLA: ${v.sla}`,
    );
  });

  console.log('\nüü° STANDARD VENDORS:');
  config.vendors.standard.forEach((v: Vendor) => {
    console.log(
      `  ${v.name.padEnd(15)} | ${v.category.padEnd(12)} | SLA: ${v.sla}`,
    );
  });

  console.log('\n='.repeat(60));
}

function showStatusPages() {
  console.log('\nüìä STATUS PAGES\n');
  for (const [vendor, url] of Object.entries(STATUS_PAGES)) {
    console.log(`  ${vendor.padEnd(15)} ‚Üí ${url}`);
  }
}

const [, , cmd] = process.argv;
switch (cmd) {
  case 'list':
    showVendors();
    break;
  case 'status':
    showStatusPages();
    break;
  default:
    console.log('Usage: npm run vendors <list|status>');
}
```
