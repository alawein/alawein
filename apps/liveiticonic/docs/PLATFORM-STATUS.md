## ğŸ’¡ Usage Example

```typescript
import {
  LaunchOrchestrator,
  AgentFactory,
  Product,
  ProductStage,
  learningEngine,
} from './src/launch-platform';

// Initialize
const orchestrator = new LaunchOrchestrator();
const agents = AgentFactory.createAllAgents();

// Register agents
agents.forEach(agent => orchestrator.registerAgent(agent));

// Plan launch
const launchPlan = await orchestrator.planLaunch(product);

// Get success prediction
const prediction = await learningEngine.predictSuccess(launchPlan);
console.log(`Success probability: ${(prediction.probability * 100).toFixed(1)}%`);

// Execute if approved
if (launchPlan.status === 'approved') {
  const result = await orchestrator.executeLaunch(launchPlan.id);

  // Learn from results
  await learningEngine.learn(experienceFromResult(result));
}
```
